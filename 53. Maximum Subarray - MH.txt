class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        
        
        max_ = nums[0]
        cur_ = 0
        for i in nums:
            if cur_ < 0: 
                cur_ = 0
            cur_ += i
            max_ = max(cur_,max_)
    
        return max_
