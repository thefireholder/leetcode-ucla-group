class Solution:
    
    def c(self,n):
        l = len(n)
        if l == 1: return n[0]
        m = int(l/2)
        if n[m-1] < n[l-1]: return self.c(n[:m])
        if n[m-1] > n[l-1]: return self.c(n[m:])
    
    def findMin(self, nums: List[int]) -> int:
        return self.c(nums)
