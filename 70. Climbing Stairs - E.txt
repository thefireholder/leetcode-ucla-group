# Runtime: 28 ms, faster than 83.78% of Python3 online submissions for Climbing Stairs.
# Memory Usage: 14 MB, less than 91.14% of Python3 online submissions for Climbing Stairs.

class Solution:
    
    saved = [0,1,2]+[0]*43
    
    def a(self,n):
        if self.saved[n] != 0: return self.saved[n]
        else:
            result = self.a(n-1) + self.a(n-2)
            self.saved[n] = result
        return result
    
    def climbStairs(self, n: int) -> int:

        return self.a(n)
